    class Solution:
    def numUniqueEmails(self, emails) -> int:
        res = set()
        for email in emails:
            cleanString = []
            i = 0
            while i < len(email) and email[i]!='@':
                if email[i] == '.':
                    i += 1
                elif email[i] == '+':
                    while email[i] != '@':
                        i += 1
                else:
                    cleanString.append(email[i])
                    i += 1
            cleanString+=email[i:]
            res.add(''.join(cleanString))
        return len(res)

    class Solution:
    def numUniqueEmails(self, emails: 'List[str]') -> 'int':
        res = set()
        for email in emails:
            name, domain = email.split('@')
            name = name.split('+')[0]
            name = name.replace(".", "")
            res.add("{}@{}".format(name, domain))

        return len(res)
