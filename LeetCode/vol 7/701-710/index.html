    class Solution:
    def insertIntoBST(self, root: TreeNode, val: int) -> TreeNode:
        p = root
        while p:
            if val > p.val:
                if p.right:
                    p = p.right
                else:
                    p.right = TreeNode(val)
                    break
            elif val < p.val:
                if p.left:
                    p = p.left
                else:
                    p.left = TreeNode(val)
                    break
        return root

    class Solution:
    def insertIntoBST(self, root: 'TreeNode', val: 'int') -> 'TreeNode':
        if not root:
            return TreeNode(val)
        if root.val>val:
            root.left=self.insertIntoBST(root.left,val)
        else:
            root.right=self.insertIntoBST(root.right,val)
        return root

    想法

    小根堆，维护一个有k大的小根堆，比堆顶数字还要小的就不放堆了，否则的就把堆顶的元素pop掉，然后重新排列堆

    重点是实现小根堆的过程。。。

def _siftdown(heap, startPos, pos):
    """
    在堆的Pos所在节点的判断，进行上浮到应该在的位置
    """
    newItem = heap[pos]
    while pos > startPos:
        parentPos = (pos - 1) >> 1
        parent = heap[parentPos]
        if newItem <= parent:
            heap[pos] = parent
            pos = parentPos
            continue
        break
    heap[pos] = newItem


def _siftup(heap, pos):
    """
    在堆的Pos的所在节点判断
    """
    endPos = len(heap)
    startPos = pos

    newItem = heap[pos]
    # why: 为什么不是现在就判断呢？
    # 选择更小的孩子
    childPos = 2*pos + 1
    while childPos < endPos:
        rightPos = childPos + 1
        if rightPos < endPos and heap[childPos] > heap[rightPos]:
            childPos = rightPos
        # 让更小的孩子上浮 小根堆
        heap[pos] = heap[childPos]
        pos = childPos
        childPos = 2*pos + 1
    # 让传入的参数所在的节点下沉到最底下
    heap[pos] = newItem
    _siftdown(heap, startPos, pos)


class KthLargest:

    def __init__(self, k, nums):
        self.heap = []
        self.k = k
        for num in nums:
            self.add(num)

    def add(self, val):
        if not self.heap or len(self.heap) < self.k:
            heapPush(self.heap, val)
        elif len(self.heap) == self.k and val > self.heap[0]:
            self.heap[0] = val
            # 5 5 10 4 9 5
            _siftup(self.heap, 0)
            # heapPush(self.heap, val)
        return self.heap[0] if len(self.heap) == self.k else None
