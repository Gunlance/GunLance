```py
class Solution:
    def numDecodings(self, s: str) -> int:
        if not s or s[0] == '0':
            return 0

        # dp[i] 表示前 i 个字符 解码它的方法总数
        dp = [0 for _ in range(len(s) + 1)]
        dp[0] = 1
        for i in range(1, len(s) + 1):
            if s[i-1] != '0':
                dp[i] += dp[i - 1]
            # '11'+ (10 |11 |12 ... |26)
            if i != 1 and s[i - 2:i] < '27' and s[i - 2:i] > '09':
                dp[i] += dp[i - 2]
        return dp[-1]

```

```py
class Solution:
    def generateTrees(self, n: int):
        if n == 0: return []
        return self._dfs(1, n)
        
    def _dfs(self, left, right):
        if left > right:
            return [None]
        res = []
        # 对于 [left,right]的每个数，都有可能作为 root
        for current_root in range(left, right + 1):
            # 左右子树都是bst树
            left_nodes = self._dfs(left,current_root-1)
            right_nodes = self._dfs(current_root + 1, right)
            for left_node in left_nodes:
                for right_node in right_nodes:
                    root = TreeNode(current_root)
                    root.left = left_node
                    root.right = right_node
                    res.append(root)
        return res
```

## 096

给定n，有多少结构上唯一的BST（二叉搜索树）存储值1 ... n？

```
Example:

Input: 3
Output: 5
Explanation:
Given n = 3, there are a total of 5 unique BST's:

   1         3     3      2      1
    \       /     /      / \      \
     3     2     1      1   3      2
    /     /       \                 \
   2     1         2                 3
```

**想法**

首先选择dp

dp[i] 表示n=i时，bst结构唯一的的数量

累加j in [1,i)
j 为左子树的数量？
dp[j]*dp[i-j-1] ?


dp[0] = 1 
dp[1] = 1
dp[2] = 2 
dp[3] = 5

### 其他

```py
class Solution:
    def numTrees(self, n: int) -> int:
        if n < 2:
            return 1
        dp = [0 for _ in range(n+1)]
        dp[0] = 1
        dp[1] = 1
        for i in range(2, n+1):
            for j in range(i):
                dp[i] += (dp[j] * dp[i - j - 1])
        return dp[n]
```

### 法一

```py
class Solution:
    def isSameTree(self, p, q):
        if p and q:
            if p.val == q.val and self.isSameTree(p.left, q.left) and self.isSameTree(p.right, q.right):
                return True
        elif p is None and q is None:
            return True

        return False
```
